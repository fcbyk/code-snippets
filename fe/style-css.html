语法&特性 ​

css
/* CSS语法：选择器 { 以及一条或多条声明 }  */
p {
  /* 声明：属性 : 属性值 ; */
  font-size: 16px;
}
css
/* 样式规则如果存在错误，解析器会选择忽略，并不会影响其他样式规则。 */
#app {
  /* 无效样式 */
  hello: red;

  /* 但不影响后面的 border 规则使用 */
  border: rgb(192, 13, 13) 1px solid;
}
css
.parent {
  color: red; /* 文本颜色会被继承 */
  font-family: Arial, sans-serif; /* 字体会被继承 */
  font-size: 18px; /* 字体大小会被继承 */
  margin: 20px; /* 不会被继承 */
  padding: 10px; /* 不会被继承 */
}

.child {
  font-size: inherit; /* 强制继承父元素的字体大小 */
  margin: inherit; /* 强制继承父元素的外边距 */
}


.child-override {
  color: inherit; /* 继承父元素的颜色 */
  margin: 0; /* 覆盖父元素的外边距 */
}
css
/* 相同权重下，后定义会覆盖先定义的样式 */
p { color: blue; }        /* 先定义 */
p { color: green; }       /* 后定义，覆盖前面的 */

/* important 设置权重最高 */
div { color: blue !important; }  /* 这条规则会覆盖其他任何color规则 */
div { color: green; }            /* 这个规则会被忽略 */

/* 
权重计算=(内联, ID, 类/属性/伪类, 元素/伪元素)
原则上，选择器越详细，权重越高
*/
div { color: blue; }        /* 0001 */
.class { color: green; }    /* 0010 */
#id { color: red; }         /* 0100 */
ul > li { list-style: none; } /* 0002 */
#hello div p { font-weight: bold;} /* 0102 */
css
/* 定义变量，必须以--开头 */
:root {
  --main-color: #3498db;  /* 主色调 */
  --font-size: 16px;      /* 字体大小 */
}

/* 使用变量 */
body {
  background-color: var(--main-color);
  font-size: var(--font-size);
  /* 可以有备选值 */
  color: var(--text-color, black);  
}

选择器 ​

html
<h1 class="c1"></h1>
<span class="c1"></span>
<h1 id="a1"></h1>

<!-- 多类样式声明，以空格分开，通过js操纵类名，可以实现样式切换 -->
<p class="one two three"></p>

<style>
  h1{ /* 标签选择器 */ }
  #a1{ /* ID选择器 */ }
  .c1{ /* 类选择器 */ }
  span.c1{ /* 标签选择器+类选择器一起使用 */ }

  /* 后面定义的样式覆盖了之前定义的样式（.three覆盖了.one） */
  .one{
    color: aquamarine;
  }
  .two{
    background-color: bisque;
  }
  .three{
    color: black;
  }
</style>
html
<div id="a1">
    <div>
        <div>
            <h1 id="hh"></h1>
            <p></p>
            <p></p>
            <span></span>
        </div>
        <p></p>
    </div>  
</div>

<style>
  h1,p{} /* 并集选择器 */
  #a1 p{} /* 后代选择器 */
  #a1 > div{} /* 子代选择器 */
  #hh + p{} /* 相邻兄弟选择器 */
  #hh ~ p{} /* 通用兄弟选择器 */
</style>
html
<h1 class="hello"></h1>
<a href="https://123.com"></a>
<a href=""></a>

<style>
  h1[class] {} /* 选择具有 class 属性的h1标签 */
  h1[class][id] {} /* 选择具有多个属性的标签 */
  a[href="https://123.com"] {} /* 选择拥有某个具体属性值的样式*/

  h2[name^="a"] {}
  h2[name$="com"] {}
  h2[name*="xiao"] {}
  h2[name~="mi"] {}
  h2[name|="abc"] {}
</style>
html
<!-- CSS 伪类 是添加到选择器的关键字，作用是，选择具有特殊状态的元素。 -->
<style>
    a:link{}
    a:visited{}
    a:hover{}
    a:active{}

    input:focus{}
    input:hover{}
    input:active{}
    input:checked + label{}
    input:valid{}

    div:target{}
    span:first-child{}
    span:nth-child(2){}

    :root{}
    :empty{}
</style>
html
<!-- 伪元素是一个附加至选择器末的关键词，允许你对被选择元素的特定部分修改样式。 -->
<style>
    p::first-line{}
    p::first-letter{}
    span::before{}
    span::after{}
</style>
