标签概述 ​

html
<标签名 属性1="值1" 属性2="值2">
    内容 or 子标签
</标签名>
html
<!-- 双标签（成对标签） -->
<div>这是内容</div> 

<!-- 单标签（自闭合标签） -->
<img src="image.jpg" />
<br/>
<input type="text" />
html
<div 
    id="unique" 
    class="style" 
    style="color:red;" 
    title="提示信息"
    myattr="不推荐的自定义属性"
    data-user-id="HTML5推荐的自定义属性，使用data-*前缀"
>
</div>
html
<hello>
    不使用JavaScript注册自定义元素
    会作为普通的行内元素存在
    可以像普通元素一样使用CSS样式
</hello>

html
<!-- 浏览器开发者工具，查看用户代理样式表，即可查看标签自带的样式 -->
<h1>标题1</h1>
<h2>标题2</h2>
<h3>标题3</h3>
<h4>标题4</h4>
<h5>标题5</h5>
<h6>标题6</h6>

<!-- h7标签的用户代理样式表为空，属于自定义表 -->
<h7>不存在标题7</h7>

<style>
    /* 把标题7设置为和标题1一样的样式 */
    h7 {
        display: block;
        font-size: 2em;
        margin-block-start: 0.67em;
        margin-block-end: 0.67em;
        margin-inline-start: 0px;
        margin-inline-end: 0px;
        font-weight: bold;
    }
</style>
html
<!-- p 指paragraph，意为段落。把文字有条理的显示出来就需要将文字分段显示。-->
<p>我是一个段落标签</p>

超链接和锚点 ​

html
<!-- 外部链接 -->
<a href="https://www.example.com">访问网站</a>

<!-- 内部页面链接 -->
<a href="about.html">关于我们</a>

<!-- 电子邮件链接 -->
<a href="mailto:example@email.com">发送邮件</a>

<!-- 电话链接 -->
<a href="tel:+18001234567">联系电话</a>

<!-- 带图片的链接 -->
<a href="https://www.example.com">
    <img src="logo.png" alt="网站logo">
</a>

<!-- 多元素链接 -->
<a href="product.html">
    <h3>产品名称</h3>
    <img src="product.jpg" alt="产品图片">
    <p>产品描述</p>
</a>
html
<!-- 新窗口打开 -->
<a href="https://www.example.com" target="_blank">在新窗口打开</a>

<!-- 下载链接 -->
<a href="document.pdf" download>下载PDF</a>

<!-- 带标题的链接 -->
<a href="https://www.example.com" title="访问我们的网站">网站链接</a>
html
<!-- 定义锚点 -->
<h2 id="section1">第一节</h2>
<p id="content1">这是第一节的内容</p>

<!-- 链接到锚点 -->
<a href="#section1">跳转到第一节</a>
<a href="#content1">查看详细内容</a>

<!-- 页面内指定位置跳转 -->
<a href="longpage.html#section3">跳转到指定章节</a>

<!-- 返回顶部链接 -->
<a href="#" style="position: fixed; bottom: 20px; right: 20px;">
    返回顶部
</a>

容器和分组标签 ​

html
<!-- div：容器标签，用于分组和布局内容 -->
<div>
    <h1>Section Title</h1>
    <p>This is a paragraph inside a div container.</p>
</div>
html
<!-- span：内联容器，用于标记或分组小块内容 -->
<p>
    This is a <span style="color: red;">highlighted</span> word in a sentence.
</p>
html
<!-- header 定义页面的头部区域，包含网站标题和导航菜单 -->
<header>
    <h1>Website Title</h1>
    <!-- nav 包含导航链接，用于引导用户访问页面的不同部分 -->
    <nav>
        <ul>
            <li><a href="#home">Home</a></li>
            <li><a href="#about">About</a></li>
            <li><a href="#contact">Contact</a></li>
        </ul>
    </nav>
</header>

<!-- section 定义主要内容区域，包含独立的内容块 -->
<section>
    <!-- article 独立的文章或内容块，语义化描述具体内容 -->
    <article>
        <h2>Article Title</h2>
        <p>This is an article with detailed content. It stands as an independent piece of information.</p>
    </article>

    <!-- aside 辅助信息区域，例如相关链接或附加内容 -->
    <aside>
        <h3>Related Links</h3>
        <ul>
            <li><a href="#link1">Related Link 1</a></li>
            <li><a href="#link2">Related Link 2</a></li>
        </ul>
    </aside>
</section>

<!-- footer 页面底部内容，通常用于版权信息或补充链接 -->
<footer>
    <p>© 2024 Your Company. All rights reserved.</p>
</footer>
