System.out.println("a ^ b = " + (a ^ b)); // 输出 6 (0101 ^ 0011 = 0110)

System.out.println("a >> 1 = " + (a >> 1)); // 输出 2 (右移一位 0101 -> 0010)
System.out.println("a >>> 1 = " + (a >>> 1)); // 输出 2 (无符号右移)

public int processNumbers(int sum) {

    for (int i = 1; i <= 100; i++) {
        if (i == 5) continue;
        if (i == 8) break;
        if (sum > 150) return sum;
        sum += i;
    }

    return sum;
}
